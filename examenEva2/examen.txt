#!/bin/bash

cuenta() {
  if [ $1 -gt $2 ]; then
    for ((i=$1; i>=$2; i--)); do
      echo $i2
    done
  elif [ $1 -lt $2 ]; then
    for ((i=$1; i<=$2; i++)); do
      echo $i
    done
  else
    echo "No se puede hacer la cuenta"
  fi
}


cuenta $1 $2

function mcd {
  param ( $a, $b)

  if ($b -gt $a) {
    echo "Error: a debe ser mayor que b";
    return
  }
  while ($b -ne 0) {
        $resto = $a % $b
        $a = $b
        $b = $resto
  }
  echo "El MCD es: $a";
}
